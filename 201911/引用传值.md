### 引用传值 & ###

    $data = ['a','b','c'];

	foreach($data as $key=>$value){
		$value = &$data[$key]
	}

	var_dump($data);

输出结果：

    array(3) { [0]=> string(1) "b" [1]=> string(1) "c" [2]=> &string(1) "c" }


解析

	/*
	 *  答案解析
	 */
	
	# 第一次 循环
	$key = 0;
	$value = 'a';
	$data[$key] = "a";
	$value = &$data[$key];
	
	# 第二次 循环
	$key = 1;
	$value = 'b';       // $data[0] = 'b';
	$data[$key] = "b";
	$value = &$data[$key];
	
	# 第三次 循环
	$key = 2;
	$value = 'c';       // $data[1] = 'c';
	$data[$key] = "c";  // $data[2] = 'c';
	$value = &$data[$key];
